- mysql 설치

```
const MySQLStore = require('express-mysql-session')(session);
```

```
app.use(session({
    secret: '1234DSFs@adf1234!@#$asd',
    resave: false,
    saveUninitialized: true,
    store: new MySQLStore({}),
}));
```
